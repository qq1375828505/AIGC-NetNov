## 步骤 2: 创建 WorkspaceManager

文件路径：`app/src/main/java/com/ai/assistance/operit/data/repository/novel/NovelWorkspaceManager.kt`

```kotlin
package com.ai.assistance.operit.data.repository.novel

import android.content.Context
import android.os.Environment
import com.ai.assistance.operit.data.model.novel.*
import kotlinx.serialization.encodeToString
import kotlinx.serialization.json.Json
import java.io.File

class NovelWorkspaceManager(private val context: Context) {

    private val json = Json { ignoreUnknownKeys = true; prettyPrint = true }
    private val root = File(Environment.getExternalStorageDirectory(), "Operit/novel_workspace")

    companion object {
        @Volatile private var INSTANCE: NovelWorkspaceManager? = null
        fun getInstance(context: Context) = INSTANCE ?: synchronized(this) {
            INSTANCE ?: NovelWorkspaceManager(context.applicationContext).also { INSTANCE = it }
        }
    }

    init {
        listOf("works", "materials/characters", "materials/settings", "materials/locations",
            "materials/factions", "materials/items", "materials/foreshadowing",
            "materials/references", "materials/todos", "memory/global", "memory/per_work",
            "templates", "exports").forEach { File(root, it).mkdirs() }
    }

    // ====== 作品 ======
    fun createWork(title: String, genre: String): NovelWork {
        val work = NovelWork(title = title, genre = genre)
        val dir = File(root, "works/${work.id}").apply { mkdirs() }
        File(dir, "chapters").mkdirs()
        File(dir, "outline").mkdirs()
        File(dir, "drafts").mkdirs()
        File(dir, "work_meta.json").writeText(json.encodeToString(work))
        File(dir, "outline/master_outline.md").writeText("# $title 总大纲\n\n## 故事背景\n\n## 主线剧情\n\n## 核心冲突\n\n## 结局走向\n")
        return work
    }

    fun getAllWorks(): List<NovelWork> = File(root, "works").listFiles()?.mapNotNull { d ->
        File(d, "work_meta.json").takeIf { it.exists() }?.let { json.decodeFromString(it.readText()) }
    }?.sortedByDescending { it.updatedAt } ?: emptyList()

    fun getWork(id: String): NovelWork? = File(root, "works/$id/work_meta.json").takeIf { it.exists() }?.let { json.decodeFromString(it.readText()) }

    fun updateWork(work: NovelWork) {
        File(root, "works/${work.id}/work_meta.json").writeText(json.encodeToString(work.copy(updatedAt = System.currentTimeMillis())))
    }

    fun deleteWork(id: String) {
        File(root, "works/$id").deleteRecursively()
        File(root, "memory/per_work/$id").deleteRecursively()
    }

    // ====== 章节 ======
    fun createChapter(workId: String, title: String, number: Int): Chapter {
        val chapter = Chapter(workId = workId, chapterNumber = number, title = title)
        File(root, "works/$workId/chapters/${String.format("%03d", number)}_${sanitize(title)}.md").writeText("")
        return chapter
    }

    fun getChapterContent(workId: String, number: Int): String =
        File(root, "works/$workId/chapters").listFiles()?.find { it.name.startsWith(String.format("%03d", number)) }?.readText() ?: ""

    fun saveChapterContent(workId: String, number: Int, content: String) {
        File(root, "works/$workId/chapters").listFiles()?.find { it.name.startsWith(String.format("%03d", number)) }?.writeText(content)
        updateWordCount(workId)
    }

    fun getChapterFiles(workId: String): List<File> = File(root, "works/$workId/chapters").listFiles()?.sortedBy { it.name } ?: emptyList()

    private fun updateWordCount(workId: String) {
        getWork(workId)?.let { work ->
            val total = File(root, "works/$workId/chapters").listFiles()?.sumOf { it.readText().length.toLong() } ?: 0
            updateWork(work.copy(currentWordCount = total))
        }
    }

    // ====== 大纲 ======
    fun getOutline(workId: String): String = File(root, "works/$workId/outline/master_outline.md").takeIf { it.exists() }?.readText() ?: ""
    fun saveOutline(workId: String, content: String) { File(root, "works/$workId/outline/master_outline.md").writeText(content) }

    // ====== 资料 ======
    fun saveMaterial(category: String, id: String, data: String) { File(root, "materials/$category/$id.json").writeText(data) }
    fun getMaterial(category: String, id: String): String? = File(root, "materials/$category/$id.json").takeIf { it.exists() }?.readText()
    fun getAllMaterials(category: String): List<Pair<String, String>> = File(root, "materials/$category").listFiles()?.map { it.nameWithoutExtension to it.readText() } ?: emptyList()
    fun deleteMaterial(category: String, id: String) { File(root, "materials/$category/$id.json").delete() }

    // ====== 记忆 ======
    fun saveMemory(id: String, data: String, workId: String? = null) {
        val dir = if (workId != null) File(root, "memory/per_work/$workId") else File(root, "memory/global")
        dir.mkdirs(); File(dir, "$id.json").writeText(data)
    }

    fun getMemory(id: String, workId: String? = null): String? {
        val dir = if (workId != null) File(root, "memory/per_work/$workId") else File(root, "memory/global")
        return File(dir, "$id.json").takeIf { it.exists() }?.readText()
    }

    // ====== 导出 ======
    fun exportWork(workId: String): String {
        val work = getWork(workId) ?: return ""
        val sb = StringBuilder()
        sb.appendLine("# ${work.title}"); sb.appendLine("题材：${work.genre}"); sb.appendLine("总字数：${work.currentWordCount}"); sb.appendLine("---\n")
        getChapterFiles(workId).forEach { sb.appendLine("## ${it.nameWithoutExtension.substringAfter("_")}\n${it.readText()}\n") }
        val out = File(root, "exports/${workId}_${System.currentTimeMillis()}.txt")
        out.writeText(sb.toString()); return out.absolutePath
    }

    private fun sanitize(name: String): String = name.replace(Regex("[^a-zA-Z0-9\\u4e00-\\u9fa5]"), "_").take(50)
}
```

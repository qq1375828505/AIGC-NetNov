## 步骤 1: 创建写作工具

文件路径：`app/src/main/java/com/ai/assistance/operit/core/tools/novel/NovelTools.kt`

```kotlin
package com.ai.assistance.operit.core.tools.novel

import com.ai.assistance.operit.core.agents.novel.NovelAgentDispatcher
import com.ai.assistance.operit.data.model.novel.*
import com.ai.assistance.operit.data.repository.novel.NovelRepository
import kotlinx.serialization.encodeToString
import kotlinx.serialization.json.Json
import org.json.JSONObject

// ====== 作品工具 ======
class CreateWorkTool(private val wm: NovelRepository) {
    val name = "create_work"
    val description = "创建新作品"
    suspend fun execute(params: JSONObject): String {
        val work = wm.createWork(params.getString("title"), params.getString("genre"))
        return "作品「${work.title}」创建成功，ID: ${work.id}"
    }
}

class ListWorksTool(private val wm: NovelRepository) {
    val name = "list_works"
    val description = "列出所有作品"
    suspend fun execute(params: JSONObject): String {
        val works = wm.getAllWorks()
        if (works.isEmpty()) return "暂无作品"
        return works.joinToString("\n") { "• ${it.title}（${it.genre}，${it.currentWordCount}字）" }
    }
}

class GetWorkTool(private val wm: NovelRepository) {
    val name = "get_work"
    val description = "获取作品详情"
    suspend fun execute(params: JSONObject): String {
        val work = wm.getWork(params.getString("work_id")) ?: return "作品不存在"
        return "标题：${work.title}\n题材：${work.genre}\n字数：${work.currentWordCount}\n状态：${work.status}"
    }
}

// ====== 章节工具 ======
class CreateChapterTool(private val wm: NovelRepository) {
    val name = "create_chapter"
    val description = "创建新章节"
    suspend fun execute(params: JSONObject): String {
        wm.createChapter(params.getString("work_id"), params.getString("title"), params.getInt("number"))
        return "章节创建成功"
    }
}

class GetChapterTool(private val wm: NovelRepository) {
    val name = "get_chapter"
    val description = "获取章节内容"
    suspend fun execute(params: JSONObject): String {
        val content = wm.getChapterContent(params.getString("work_id"), params.getInt("number"))
        return if (content.isEmpty()) "章节内容为空" else content
    }
}

class SaveChapterTool(private val wm: NovelRepository) {
    val name = "save_chapter"
    val description = "保存章节内容"
    suspend fun execute(params: JSONObject): String {
        wm.saveChapterContent(params.getString("work_id"), params.getInt("number"), params.getString("content"))
        return "章节保存成功"
    }
}

// ====== 角色工具 ======
class CreateCharacterTool(private val wm: NovelRepository) {
    val name = "create_character"
    val description = "创建角色卡"
    suspend fun execute(params: JSONObject): String {
        val character = Character(
            name = params.getString("name"),
            position = params.optString("position", ""),
            appearance = params.optString("appearance", ""),
            personality = params.optString("personality", ""),
            background = params.optString("background", "")
        )
        wm.saveMaterial("characters", character.id, Json.encodeToString(character))
        return "角色「${character.name}」创建成功，ID: ${character.id}"
    }
}

class ListCharactersTool(private val wm: NovelRepository) {
    val name = "list_characters"
    val description = "列出所有角色"
    suspend fun execute(params: JSONObject): String {
        val chars = wm.getAllMaterials("characters").map { Json.decodeFromString<Character>(it.second) }
        if (chars.isEmpty()) return "暂无角色"
        return chars.joinToString("\n") { "• ${it.name}（${it.position}）" }
    }
}

// ====== 大纲工具 ======
class GetOutlineTool(private val wm: NovelRepository) {
    val name = "get_outline"
    val description = "获取大纲内容"
    suspend fun execute(params: JSONObject): String {
        val outline = wm.getOutline(params.getString("work_id"))
        return if (outline.isEmpty()) "大纲为空" else outline
    }
}

class SaveOutlineTool(private val wm: NovelRepository) {
    val name = "save_outline"
    val description = "保存大纲内容"
    suspend fun execute(params: JSONObject): String {
        wm.saveOutline(params.getString("work_id"), params.getString("content"))
        return "大纲保存成功"
    }
}

// ====== AI写作工具 ======
class PolishTextTool(private val dispatcher: NovelAgentDispatcher) {
    val name = "polish_text"
    val description = "精修文本"
    suspend fun execute(params: JSONObject): String {
        return dispatcher.dispatch("请精修以下文本：\n${params.getString("text")}")
    }
}

class CheckPleasureTool(private val dispatcher: NovelAgentDispatcher) {
    val name = "check_pleasure"
    val description = "检查爽点密度"
    suspend fun execute(params: JSONObject): String {
        return dispatcher.dispatch("请检查以下文本的爽点密度：\n${params.getString("text")}")
    }
}

class GenerateOutlineTool(private val dispatcher: NovelAgentDispatcher) {
    val name = "generate_outline"
    val description = "生成大纲"
    suspend fun execute(params: JSONObject): String {
        return dispatcher.dispatch("请为${params.getString("genre")}题材生成大纲，关键词：${params.optString("keywords", "")}")
    }
}

class GenerateCharacterTool(private val dispatcher: NovelAgentDispatcher) {
    val name = "generate_character"
    val description = "生成角色设定"
    suspend fun execute(params: JSONObject): String {
        return dispatcher.dispatch("请为${params.optString("genre", "")}小说生成一个${params.getString("position")}角色的详细设定")
    }
}

// ====== 导出工具 ======
class ExportWorkTool(private val wm: NovelRepository) {
    val name = "export_work"
    val description = "导出作品"
    suspend fun execute(params: JSONObject): String {
        val path = wm.exportWork(params.getString("work_id"))
        return if (path.isEmpty()) "导出失败" else "导出成功：$path"
    }
}

// ====== 工具注册表 ======
object NovelToolRegistry {
    fun createTools(wm: NovelRepository, dispatcher: NovelAgentDispatcher): Map<String, Any> {
        return mapOf(
            "create_work" to CreateWorkTool(wm),
            "list_works" to ListWorksTool(wm),
            "get_work" to GetWorkTool(wm),
            "create_chapter" to CreateChapterTool(wm),
            "get_chapter" to GetChapterTool(wm),
            "save_chapter" to SaveChapterTool(wm),
            "create_character" to CreateCharacterTool(wm),
            "list_characters" to ListCharactersTool(wm),
            "get_outline" to GetOutlineTool(wm),
            "save_outline" to SaveOutlineTool(wm),
            "polish_text" to PolishTextTool(dispatcher),
            "check_pleasure" to CheckPleasureTool(dispatcher),
            "generate_outline" to GenerateOutlineTool(dispatcher),
            "generate_character" to GenerateCharacterTool(dispatcher),
            "export_work" to ExportWorkTool(wm)
        )
    }
}
```

# AI 工具注册

> **状态：❌ 未实施** | 统一决策见 `Operit网文写作改造计划.md` 5.5 节与附录 B
> - 工具数量：**48 个**，分类见附录 B。
> - 工具优先在 **ToolPkg `src/packages/*.ts`** 中实现，通过 `ToolPkg.Registry` 注册为子包工具（如 `novelide:create_character`）。
> - 数据访问统一使用 `NovelRepository` + Room DAO（`NovelWorkspaceManager` 已废弃）。
> - 关键：function-calling JSON 为主，Operit 需要时提供 XML fallback。

## 工具示例

```kotlin
class CreateCharacterTool(
    private val novelRepository: NovelRepository
) : AITool {
    override val name = "create_character"
    override val description = "创建新角色卡"
    
    override suspend fun execute(params: JSONObject): ToolResult {
        val name = params.getString("name")
        val position = params.optString("position", "")
        val appearance = params.optString("appearance", "")
        val personality = params.optString("personality", "")
        val background = params.optString("background", "")
        
        val character = Character(
            name = name,
            position = position,
            appearance = appearance,
            personality = personality,
            background = background
        )
        
        novelRepository.saveMaterial(
            "characters",
            character.id,
            Json.encodeToString(character)
        )
        
        return ToolResult.success("角色「$name」创建成功，ID: ${character.id}")
    }
    
    override val parameters = listOf(
        ToolParameter("name", "角色名", true),
        ToolParameter("position", "定位", false),
        ToolParameter("appearance", "外貌", false),
        ToolParameter("personality", "性格", false),
        ToolParameter("background", "背景", false)
    )
}
```
